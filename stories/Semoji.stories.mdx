**Switchable Twitter/GitHub emojis**.

## Installation

```sh
npm install --save switch-emoji
```

## Usage

```js
import { Semoji } from "switch-emoji";

export default function GHEmoji(props) {
  return <Semoji {...props} source="GitHub" />;
}
```
